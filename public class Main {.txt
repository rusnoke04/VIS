public class Main {
    public static void main(String[] args) {
        // Hello, World! and sum of numbers
        System.out.println("Hello, World!");
        
        int sum = 0;
        for (int i = 1; i <= 5; i++) {
            sum += i;
        }
        
        System.out.println("Sum of first 5 numbers: " + sum);
    }
}
